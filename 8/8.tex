\begin{problem}{2}
    Let $a_{ij}$ be the number in the $i$th row and $j$th column of the array
    $$-1 \qquad 0 \qquad 0 \qquad 0 \qquad \dots$$
    $$\frac{1}{2} \quad -1 \qquad 0 \qquad 0 \qquad \dots$$
    $$\frac{1}{4} \qquad \frac{1}{2} \quad -1 \qquad 0 \qquad \dots$$
    $$\frac{1}{8} \qquad \frac{1}{4} \qquad \frac{1}{2} \quad -1 \qquad \dots$$
    $$\dots\dots\dots\dots\dots\dots\dots\dots\dots\dots\dots$$
    so that 
    \begin{equation}
        a_{ij}=
        \begin{cases}
                0, & (i < j) \\
                -1, & (i=j) \\
                2^{j-1}, & (i>j) \\
        \end{cases}
    \end{equation}
    Prove that 
    $$\sum_i\sum_ja_{ij}=-2, \qquad \sum_j\sum_ia_{ij}=0$$
\end{problem}
\begin{solution}{}
    Summing vertically first 
    $$\sum_j\sum_ia_{ij}$$
    we have that 
    $$-1 + \sum_{k=1}^{\infty}2^{-k} = -1 + 1 = 0$$
    for each $i$ thus $\sum_j\sum_ia_{ij}=0$.\\
    Summing horizontally first
    $$\sum_j\sum_ia_{ji}$$
    we have
    $$\sum_{k=0}^{\infty}-2^{-k} = -1 - \sum_{k=1}^{\infty}2^{-k}=-1-1=-2$$
    \qed
\end{solution}

\begin{problem}{4}
    Prove that following limit relations:
    $$(a) \lim_{x\to 0}\frac{b^x - 1}{x} = \log b$$
    $$(b) \lim_{x\to 0}\frac{\log(1+x)}{x} = 1$$
    $$(c) \lim_{x\to 0}(1+x)^{1/x} = e$$
    $$(d) \lim_{n\to \infty}\bigg(1+\frac{x}{n}\bigg)^n = e^x$$
\end{problem}
\begin{solution}{}
    (a) Using L'hospital's rule, the denominator becomes 1 and numerator $\log b$.\\
    (b) Same as in a), we get $\lim_{x\to 0}\frac{1}{1+x}=1$\\
    (c) 
    $$\lim_{x\to 0}(1+x)^{1/x} = \lim_{x\to 0} e^{\frac{\log(1+x)}{x}}$$
    $$= e^{\lim_{x\to 0}\frac{\log(1+x)}{x}} = e^1 = e$$
    (d) 
    $$\lim_{n\to \infty}\bigg(1+\frac{x}{n}\bigg)^n=
    \lim_{n\to \infty}\bigg(\bigg(1+\frac{x}{n}\bigg)^{\frac{n}{x}}\bigg)^x$$
    $$=\bigg(\lim_{y\to 0}\bigg(1+y\bigg)^{\frac{1}{y}}\bigg)^x$$
    $$=e^x$$
    \qed
\end{solution}
